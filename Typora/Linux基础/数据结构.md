void ReverseArr(int arr[], int len) {
    int carr[len];
    for (int i=0; i<len; ++i) {
        carr[len-i-1] = arr[i];
    }
    for (int i=0; i<len; ++i) {
        arr[i] = carr[i];
    }
}
